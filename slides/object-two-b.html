let obj1 = Object.create({
      bar: 'baz'
    });

    obj1.foo = 'bar';

    let o1Proto = Object.getPrototypeOf(obj1);
    let obj2 = Object.assign(
      Object.create(o1Proto),
      obj1);

    obj2.bar // 'baz'
Only copies own properties, not inherited.
If we want it to have the same prototype, we can assign to a new object with the same prototype
