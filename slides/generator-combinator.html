function* even(iterable) {
      let iter = iterable[Symbol.iterator]();
      let next = iter.next();
      while(!next.done) {
        if (next.value % 2 === 0) {
          yield next;
        }
        next = iter.next();
      }
    }

    function* take(n, iterable) {
      let iter = iterable[Symbol.iterator]();
      while(n--) {
        yield (iter.next()).value;
      }
    }
