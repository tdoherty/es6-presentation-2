Array.from('foo');
    // ['f', 'o', 'o']

    Array.from([1,2,3], x => x * x);
    // [1, 4, 9]

    function f() {
      return Array.from(arguments);
    }
    f(1, 2, 3); //[1, 2, 3]

    Array.from({length: 5}, (v, i) => i);
    // [0, 1, 2, 3, 4]
Works with array-like objects, as long as has a length property
Takes an optional map function to modify the source
