let arr = [ {name: 'foo', val: '1'},
                {name: 'bar', val: '2'},
                {name: 'baz', val: '3'} ];

    // ES5
    function find(arr, prop, value) {
      var len == arr.length;
      while(len--) {
        if (arr[len][prop] === value) {
          return arr[len];
        }
      }
    }

    // ES6
    arr.find(e => e.val === '2');
Replaces _.find() for arrays, though without the overloads
provided by lodash. Still, native!
