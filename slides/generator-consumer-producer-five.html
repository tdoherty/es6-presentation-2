function* foo(x) {
      var y = yield (x + 1);
      var z = 3 (y + 1); <-resume
      return (x + y + z);
    }

    let iter = foo(1);

    // note: no param to `next()` here
    iter.next();  // { value: 2, done: false }
    iter.next(2); // { value: 3, done: false }
    iter.next(3); // { value: 6, done: true }
