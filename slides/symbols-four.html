let Thing = (function () {
      let nameSymbol = Symbol();
      return class Thing {
        constructor(name) {
          this[nameSymbol] = name;
        }
        get name() {
          return this[nameSymbol];
        }
      };
    })();

    let it = new Thing('foo');
    it.name           // 'foo'
    it.name = 'bar';  // TypeError
    Object.getOwnPropertyNames(it) // []
